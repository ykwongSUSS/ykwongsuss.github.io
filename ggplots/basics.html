1 Scatterplot

# 1. Scatter
library(tidyverse)
ggplot(df, aes(x=girth, y=height, color=sex))+
  geom_point(size=2)+
  labs(
    title="Scatter Plot",
    subtitle="Date: 28 Jul 2024",
    x = "Girth(cm)",
    y = "Height(cm)"
  )

2 ScatterLines

# 2. ScatterLines
ggplot(df)+
  xlim(0,100)+
  ylim(0,80)+
  # girth
  geom_point(x=age,y=girth,shape=21, size=4,fill="orange",alpha=0.4)+
  geom_line(aes(x=age,y=girth,color="girth"),show.legend=FALSE)+
  # height
  geom_point(x=age,y=height,shape=21, size=4,fill="yellow")+
  geom_line(aes(x=age,y=height,color="height"),show.legend=FALSE)+
  # color
  scale_color_manual(values=c("red","blue"))+
  ggtitle("ScatterLines")+
  ylab("girth-height")+
  theme_bw()

3 BarPlot

# barplot
direction = c('N','S','E','W','W','E','W','E','E','E','E','S','S','N','N','W','E','E','N','N','N','S','E','W','W','E','W','E','E','E')
speed = rnorm(30, 5, 0.2)
dfWind <- data.frame(direction,speed)
str(dfWind)
'data.frame':   30 obs. of  2 variables:
 $ direction: chr  "N" "S" "E" "W" ...
 $ speed    : num  4.97 5.16 4.83 5.03 4.98 ...
# indicate x
# counting on y is automatic

ggplot(dfWind,aes(x=direction, fill=direction))+
  geom_bar(color="gray")+
  labs(
    title="BarPlot",
    subtitle="auto counting"
  )

4 ColumnPlot

school <- c("NSHD","SHBS","SBIZ","LAW")
numSG <- c(68,140,90,55)
df2 <- data.frame(x=school,y=numSG)
df2
     x   y
1 NSHD  68
2 SHBS 140
3 SBIZ  90
4  LAW  55
# both x and y must be indicated
ggplot(df2, aes(x=school, y=numSG, fill=school))+
  labs(
    title="ColumnPlot",
    subtitle="yValues")+
  geom_col(color="gray")+
  scale_fill_manual(values=c("lightgreen","yellow","lightblue","purple"))+
  geom_text(aes(label=numSG),nudge_y=10)+
  ylim(0,180)+
  theme_bw()
