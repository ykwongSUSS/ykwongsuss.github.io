library(tidyverse)

ID <- str_c("SD",101:110)

3. Define the “cuts”

like a lookup table

gletters <- c("F","D","D+","C","C+","B-","B","B+","A-","A","A+")
cutScores <- c(0,40,44,49,54,59,64,69,74,79,84,100)

ltrGrades <- cut(df$TotScore, cutScores,right=TRUE, labels=gletters)
df$TotScore
 [1] 56.0 28.4 67.6 85.8 67.4 64.0 83.0 50.0 63.2 61.6
ltrGrades
 [1] C+ F  B  A+ B  B- A  C  B- B-
Levels: F D D+ C C+ B- B B+ A- A A+
