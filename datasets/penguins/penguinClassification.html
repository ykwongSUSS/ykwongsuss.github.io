glm_spec <- logistic_reg()%>%
  set_engine("glm") 

rf_spec <- rand_forest() %>%
  set_mode("classification") %>%
  set_engine("ranger")
penguin_wf <- workflow() %>%
  add_formula(sex ~ .)
